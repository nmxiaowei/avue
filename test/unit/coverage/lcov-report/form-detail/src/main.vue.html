export default {
  name: "AvueFormDetail",
  props: {
    option: {
      type: Object,
      default: () => { }
    },
    value: {}
  },
  computed: {
    labelPostion: function () {
      if (this.option.labelPostion) {
        return "is-" + this.option.labelPostion;
      }
      return 'is-left'
    },
    columnOption () {
      return this.option.option || []
    }
  },
  watch: {
    value: {
      handler (n) {
        this.form = n;
      },
      deep: true
    },
  },
  data () {
    return {
      form: {}
    }
  },
  created () {
    this.formInit();
  },
  methods: {
    formInit () {
      this.form = this.value;
    }
  }
}
